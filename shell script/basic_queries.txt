SELECT QUANTITY
FROM PRODUCT
WHERE TITLE = “Pepe”;

SELECT DISTINCT DELIVERY_STATUS
FROM PRODUCT
WHERE TITLE = "Mail";

SELECT DISTINCT DIRECTOR
FROM MOVIE
WHERE lead_actor = “Donald Trump”;

SELECT DISTINCT LEAD_ARTIST
FROM MUSIC
WHERE PRODUCER = “Rye”;

SELECT DISTINCT EMPLOYEE_ID
FROM EMPLOYEE
WHERE EMPLOYEE_NAME = “Bill Gates”;

SELECT DISTINCT MANUFACTURER_ADDRESS
FROM MANUFACTURER
WHERE MANUFACTURER_NAME = “Ryerson”;

SELECT DISTINCT MANAGER
FROM STORE
WHERE STORE_NAME = “Indico Store”;

SELECT EMPLOYEE_NAME, EMPLOYEE_ID
FROM EMPLOYEE
ORDER BY EMPLOYEE_ID DESC;

CREATE VIEW Store_Info(StoreName StoreAddress,StoreNumber) AS
SELECT STORE_NAME, STORE_ADDRESS, STORE_NUMBER
FROM STORE;

CREATE VIEW Employee_Info(EmployeeName, EmployeeID) AS
SELECT EMPLOYEE_NAME, EMPLOYEE_ID 
FROM EMPLOYEE;
